3    # Investment calculation
4    def investment_calculation(amount, interest_rate, years, interest_type):
5        interest_rate = interest_rate / 100
6        if interest_type.lower() == 'simple':
7            return amount * (1 + interest_rate * years)
8        elif interest_type.lower() == 'compound':
9            return amount * math.pow((1 + interest_rate), years)
10       else:
11           return "Invalid interest type. Please enter either 'simple' or 'compound'"

13   # Bond repayment calculation
14   def bond_repayment_calculation(present_value, interest_rate, months):
15       monthly_interest_rate = (interest_rate / 100) / 12
16       return (monthly_interest_rate * present_value) / (1 - (1 + monthly_interest_rate)**(-months))
